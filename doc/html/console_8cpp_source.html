00032 int console::findArguments 
00033 (int argc, const char** argv, const char* argument_name) {
00034     for (int i = 1; i < argc; ++i)
00035         // Search for the string
00036         if (strcmp (argv[i], argument_name) == 0)
00037             return (i);
00038 
00039     return (-1);
00040 }
